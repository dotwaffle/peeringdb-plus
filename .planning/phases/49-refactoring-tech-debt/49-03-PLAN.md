```go
package database

import (
    "testing"
    "path/filepath"
)

func TestOpen_Success(t *testing.T) {
    t.Parallel()
    dbPath := filepath.Join(t.TempDir(), "test.db")
    client, db, err := Open(dbPath)
    if err != nil { t.Fatalf(...) }
    defer client.Close()
    defer db.Close()
    // Verify non-nil
}

func TestOpen_Pragmas(t *testing.T) {
    t.Parallel()
    dbPath := filepath.Join(t.TempDir(), "test.db")
    _, db, err := Open(dbPath)
    if err != nil { t.Fatalf(...) }
    defer db.Close()

    tests := []struct {
        name   string
        pragma string
        want   string
    }{
        {"journal_mode", "PRAGMA journal_mode", "wal"},
        {"foreign_keys", "PRAGMA foreign_keys", "1"},
        {"busy_timeout", "PRAGMA busy_timeout", "5000"},
    }
    for _, tt := range tests {
        t.Run(tt.name, func(t *testing.T) {
            var got string
            err := db.QueryRow(tt.pragma).Scan(&got)
            if err != nil { t.Fatalf(...) }
            if got != tt.want { t.Errorf(...) }
        })
    }
}

func TestOpen_PoolConfig(t *testing.T) {
    t.Parallel()
    dbPath := filepath.Join(t.TempDir(), "test.db")
    _, db, err := Open(dbPath)
    if err != nil { t.Fatalf(...) }
    defer db.Close()
    stats := db.Stats()
    if stats.MaxOpenConnections != 10 { t.Errorf("MaxOpenConnections = %d, want 10", stats.MaxOpenConnections) }
}
```

Use `t.Parallel()` on all tests per GO-T-3. Use `t.TempDir()` for automatic cleanup.
