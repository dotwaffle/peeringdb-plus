Land per-request heap-delta telemetry per D-06. Three concerns:

1. **Sampler** — `internal/pdbcompat/telemetry.go` exposes `recordResponseHeapDelta(ctx, endpoint, entity, startHeapKiB)`. Called via `defer` at the top of serveList. `runtime.ReadMemStats` is STW (~microseconds at our heap size) so D-06 permits it once per request but never per row — enforced by calling the sampler once at entry and once at exit (via defer).

    Create `internal/pdbcompat/telemetry.go`:
    ```
    package pdbcompat

    import (
        "context"
        "runtime"

        "go.opentelemetry.io/otel/attribute"
        "go.opentelemetry.io/otel/metric"
        "go.opentelemetry.io/otel/trace"

        pdbotel "github.com/dotwaffle/peeringdb-plus/internal/otel"
    )

    // memStatsHeapInuseKiB samples runtime.MemStats.HeapInuse in KiB.
    // ReadMemStats is STW (~µs at our heap size); safe once per request
    // per D-06 but NEVER per row.
    func memStatsHeapInuseKiB() int64 {
        var ms runtime.MemStats
        runtime.ReadMemStats(&ms)
        const maxInt64 = uint64(1<<63 - 1)
        if ms.HeapInuse >= maxInt64 {
            return int64(maxInt64) / 1024
        }
        return int64(ms.HeapInuse) / 1024
    }

    // recordResponseHeapDelta samples HeapInuse, computes delta vs
    // startKiB, and emits both the OTel span attribute and the
    // Prometheus histogram observation. Invoked via defer at the top of
    // serveList so every terminal path (200, 413, 400, 500) fires it
    // exactly once.
    //
    // Negative deltas are clamped to 0 — GC cycles between entry and
    // exit can legitimately reduce HeapInuse, and a negative histogram
    // sample is not meaningful for "how much heap did this request cost".
    func recordResponseHeapDelta(ctx context.Context, endpoint, entity string, startKiB int64) {
        endKiB := memStatsHeapInuseKiB()
        delta := endKiB - startKiB
        if delta < 0 {
            delta = 0
        }
        // OTel span attribute.
        if span := trace.SpanFromContext(ctx); span.SpanContext().IsValid() {
            span.SetAttributes(attribute.Int64("pdbplus.response.heap_delta_kib", delta))
        }
        // Prometheus histogram — nil-safe if metric registration failed
        // (don't crash the request path over a telemetry hiccup).
        if pdbotel.ResponseHeapDeltaKiB != nil {
            pdbotel.ResponseHeapDeltaKiB.Record(ctx, delta,
                metric.WithAttributes(
                    attribute.String("endpoint", endpoint),
                    attribute.String("entity", entity),
                ),
            )
        }
    }
    ```
