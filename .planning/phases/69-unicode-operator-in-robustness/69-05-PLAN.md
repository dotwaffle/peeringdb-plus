<rationale>
**Build-tag-gated bench toggle (B1 + W3 fix).** An earlier draft of this plan proposed a `disableShadowRoute` package-level `bool` living inside production `internal/pdbcompat/filter.go`, toggled by the benchmark. That violates GO-CC-3 (no package-level mutable production state) AND required filter.go to be listed in `files_modified`, which it was not — the undeclared-file edit is a blocker. The fix is to move the toggle into `internal/pdbcompat/filter_bench.go` guarded by `//go:build bench`, and gate the benchmarks themselves with `//go:build bench`. Non-bench production builds never see either file; benchmarks run under `go test -tags=bench -bench=...`. This removes the edit to filter.go entirely, aligns with GO-CC-3, and is cleaner than a runtime flag.

**Fuzz file naming (B3 fix).** CONTEXT.md D-07 references `filter_fuzz_test.go` as a filename hint. The existing repo has `fuzz_test.go`. The plan keeps `fuzz_test.go` — no rename — to preserve git history. The D-07 reference is interpreted as a filename hint, not a rename mandate. `files_modified` lists ONLY `fuzz_test.go` to match the action.
</rationale>

    ```go
    // +build bench

    // Package pdbcompat: bench-only helpers for Phase 69 index-decision
    // benchmarks. See bench_test.go. NEVER compiled in production builds;
    // guarded by //go:build bench so non-bench `go build ./...` never sees
    // these symbols (GO-CC-3: no package-level mutable production state).
    package pdbcompat

    import (
        "entgo.io/ent/dialect/sql"

        "github.com/dotwaffle/peeringdb-plus/internal/unifold"
    )

    // directContainsPredicate reproduces the Phase 68 non-shadow path for
    // benchmark comparison against the Phase 69 shadow path. Used ONLY by
    // bench_test.go under -tags=bench. Reading the shadow-route production
    // code is out of scope for this plan — we measure the two predicate
    // shapes directly here.
    func directContainsPredicate(field, value string) func(*sql.Selector) {
        return sql.FieldContainsFold(field, value)
    }

    // shadowContainsPredicate reproduces the Phase 69 shadow path with
    // unifold.Fold applied to the value and the _fold suffix on the column.
    func shadowContainsPredicate(field, value string) func(*sql.Selector) {
        return sql.FieldContainsFold(field+"_fold", unifold.Fold(value))
    }
    ```

    **Step B — Write the benchmark, also `//go:build bench`-gated.**

    Create `internal/pdbcompat/bench_test.go`:

    ```go
    //go:build bench
    // +build bench

    package pdbcompat

    import (
        "context"
        "fmt"
        "net/url"
        "testing"

        "github.com/dotwaffle/peeringdb-plus/ent"
        "github.com/dotwaffle/peeringdb-plus/internal/peeringdb"
        "github.com/dotwaffle/peeringdb-plus/internal/testutil"
    )

    // seedNetworks bulk-creates N networks with deterministic names so the
    // benchmark exercises the LIKE %zurich% path against a realistic text
    // column. IDs start at 100 to avoid collision with seed/Full.
    func seedNetworks(tb testing.TB, client *ent.Client, n int) {
        tb.Helper()
        ctx := context.Background()
        tx, err := client.Tx(ctx)
        if err != nil { tb.Fatalf("tx: %v", err) }
        defer tx.Rollback() //nolint:errcheck

        builders := make([]*ent.NetworkCreate, n)
        for i := 0; i < n; i++ {
            name := fmt.Sprintf("Network-%d-Zürich", i)
            builders[i] = tx.Network.Create().
                SetID(100+i).
                SetOrgID(1).
                SetName(name).
                SetNameFold(fmt.Sprintf("network-%d-zurich", i)). // matches unifold.Fold
                SetASN(uint64(65000+i)).
                SetStatus("ok")
        }
        if err := tx.Network.CreateBulk(builders...).Exec(ctx); err != nil {
            tb.Fatalf("bulk create: %v", err)
        }
        if err := tx.Commit(); err != nil { tb.Fatalf("commit: %v", err) }
    }

    // benchNameContains runs the query path for each of (direct, shadow) on
    // the same HEAD commit without editing production code. The two
    // predicate builders come from filter_bench.go (also //go:build bench).
    func benchNameContains(b *testing.B, rows int, shadow bool) {
        client := testutil.SetupClient(b)
        seedNetworks(b, client, rows)

        predBuilder := directContainsPredicate
        if shadow {
            predBuilder = shadowContainsPredicate
        }
        ctx := context.Background()
        _ = url.Values{} // placeholder for extending the bench to full ParseFilters path

        b.ResetTimer()
        for i := 0; i < b.N; i++ {
            pred := predBuilder("name", "Zurich")
            _, err := client.Network.Query().Where(pred).All(ctx)
            if err != nil { b.Fatalf("query: %v", err) }
        }
    }

    func BenchmarkNameContains_100_Direct(b *testing.B)   { benchNameContains(b, 100, false) }
    func BenchmarkNameContains_100_Shadow(b *testing.B)   { benchNameContains(b, 100, true) }
    func BenchmarkNameContains_10000_Direct(b *testing.B) { benchNameContains(b, 10_000, false) }
    func BenchmarkNameContains_10000_Shadow(b *testing.B) { benchNameContains(b, 10_000, true) }
    ```

    Note: the reference type of `predBuilder` is `func(field, value string) func(*sql.Selector)`; Go infers it from `directContainsPredicate`. If the Go compiler complains about implicit conversion, add an explicit type alias at the top of the file.
