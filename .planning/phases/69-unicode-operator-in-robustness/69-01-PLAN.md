    ```go
    // Package unifold folds Unicode strings into a normalised ASCII-lowercase
    // form suitable for diacritic-insensitive equality/substring matching.
    //
    // The pipeline is: hand-mapped ligature substitution → NFKD normalisation
    // (golang.org/x/text/unicode/norm) → drop combining marks (unicode.Mn) →
    // ToLower. The hand map covers non-decomposable diacritics that NFKD
    // alone leaves intact (ß, æ, ø, ł, þ, đ, and their upper-case variants).
    //
    // Scope: reproduces upstream PeeringDB's `unidecode.unidecode(v)` behaviour
    // (rest.py:576) closely enough for filter-value matching in the pdbcompat
    // layer. It is NOT a full Unicode-to-ASCII transliteration library —
    // CJK, Arabic, Hebrew, and other non-Latin scripts pass through untouched
    // so that foreign-language substring matches still work against the
    // folded DB column.
    package unifold

    import (
        "strings"
        "unicode"

        "golang.org/x/text/unicode/norm"
    )

    // foldMap covers non-decomposable ligatures and stroke-letters that NFKD
    // leaves unchanged. Upper and lower variants both mapped so ToLower order
    // never races the substitution.
    //
    // Sourced from upstream `unidecode` behaviour for these specific code
    // points; keep list tight — expanding beyond the 6 pairs decided in D-02
    // requires a new decision.
    var foldMap = map[rune]string{
        'ß': "ss", // U+00DF LATIN SMALL LETTER SHARP S
        'ẞ': "ss", // U+1E9E LATIN CAPITAL LETTER SHARP S
        'æ': "ae",
        'Æ': "ae",
        'ø': "o",
        'Ø': "o",
        'ł': "l",
        'Ł': "l",
        'þ': "th",
        'Þ': "th",
        'đ': "d",
        'Đ': "d",
    }

    // Fold normalises s for case-insensitive, diacritic-insensitive matching.
    // Total function: any UTF-8 input (including invalid bytes) returns a
    // string without panicking. Empty input returns "".
    func Fold(s string) string {
        if s == "" {
            return ""
        }
        // Phase 1: hand-map non-decomposable ligatures.
        // Cheap bail: if no rune in s is in foldMap AND all bytes are
        // ASCII lowercase, return s unchanged (fast path).
        if asciiLowerFastPath(s) {
            return s
        }
        var sub strings.Builder
        sub.Grow(len(s))
        for _, r := range s {
            if rep, ok := foldMap[r]; ok {
                sub.WriteString(rep)
                continue
            }
            sub.WriteRune(r)
        }
        // Phase 2: NFKD decompose, then drop combining marks (Mn category).
        decomposed := norm.NFKD.String(sub.String())
        var out strings.Builder
        out.Grow(len(decomposed))
        for _, r := range decomposed {
            if unicode.Is(unicode.Mn, r) {
                continue
            }
            out.WriteRune(unicode.ToLower(r))
        }
        return out.String()
    }

    // asciiLowerFastPath returns true when every byte of s is in [a-z0-9 -_].
    // Conservative: any non-ASCII byte or uppercase letter forces the full
    // pipeline. Strings passing this check are idempotent under Fold.
    func asciiLowerFastPath(s string) bool {
        for i := 0; i < len(s); i++ {
            c := s[i]
            if c >= 0x80 {
                return false
            }
            if c >= 'A' && c <= 'Z' {
                return false
            }
        }
        return true
    }
    ```

    Notes on the design, per D-02:
    - NFKD decomposes `é` → `e + U+0301`; the `unicode.Is(unicode.Mn, r)` guard drops the combining acute. This handles 95% of diacritics with zero maintenance.
    - The hand map MUST run BEFORE NFKD so that `ß` → `ss` applies before `ToLower` on the NFKD output. Running it after NFKD would still work but is less readable.
    - The fast path is not required for correctness; it avoids allocating a strings.Builder for the common all-ASCII-lowercase case (e.g. the `_fold` column populated by the sync worker reading back through Fold).
    - `unicode.ToLower` handles ASCII A-Z → a-z plus a few Unicode cases. It's idempotent, so applying to already-lowercase CJK is a no-op.
    - No separate `doc.go` file — the `package unifold` comment above is the godoc surface per Go convention (`go doc internal/unifold` emits the same text either way).
