**Global htmx error handler in layout.templ (per UI-04, Pattern 2 from RESEARCH.md):**

Add a `<script>` block in layout.templ, after the existing keyboard navigation IIFE (before `</body>`). This script attaches a global `htmx:afterRequest` event listener that detects failed collapsible section fetches:

```javascript
// htmx error handling for collapsible sections (UI-04).
// On failed fetch inside a <details> element, replaces "Loading..." with
// an error message and retry button.
(function() {
    document.body.addEventListener('htmx:afterRequest', function(evt) {
        if (!evt.detail.successful && evt.detail.elt && evt.detail.elt.closest('details')) {
            var el = evt.detail.elt;
            var url = el.getAttribute('hx-get');
            // Clear the loading indicator
            el.textContent = '';
            // Build error UI using DOM API (avoids XSS from string-based markup)
            var wrapper = document.createElement('div');
            wrapper.className = 'px-4 py-3 text-center';
            var msg = document.createElement('span');
            msg.className = 'text-red-400 text-sm';
            msg.textContent = 'Failed to load.';
            var btn = document.createElement('button');
            btn.className = 'text-emerald-400 hover:text-emerald-300 text-sm underline ml-2';
            btn.textContent = 'Retry';
            btn.setAttribute('hx-get', url);
            btn.setAttribute('hx-target', 'closest div');
            btn.setAttribute('hx-swap', 'innerHTML');
            wrapper.appendChild(msg);
            wrapper.appendChild(btn);
            el.appendChild(wrapper);
            // Initialize htmx on the new retry button so hx-get works
            htmx.process(el);
        }
    });
})();
```

Place this in its own `<script>` block, NOT inside the DOMContentLoaded handler (htmx events fire on body regardless of DOMContentLoaded).

**No changes needed to CollapsibleSection/CollapsibleSectionWithBandwidth in detail_shared.templ** for this task -- the error handling is entirely client-side via the global event listener. However, if the inner `<div>` with `hx-get` needs a distinguishing attribute for targeting, the existing structure is sufficient since `evt.detail.elt` is already the `<div>` that has `hx-get`.
