    func TestDefaultOrdering_Pdbcompat(t *testing.T) {
        t.Parallel()
        cases := []struct {
            name string
            seed func(t *testing.T, client *ent.Client) []int // returns expected id order

                ctx := context.Background()
                client := testutil.SetupClient(t)
                expectedIDs := tc.seed(t, client)
                srv := httptest.NewServer(pdbcompat.Router(client /* or however registered */))
                defer srv.Close()
                resp, err := http.Get(srv.URL + tc.path)
                if err != nil { t.Fatal(err) }

                var payload struct{ Data []struct{ ID int `json:"id"` } `json:"data"` }

                gotIDs := make([]int, len(payload.Data))
                for i, r := range payload.Data { gotIDs[i] = r.ID }
                if !reflect.DeepEqual(gotIDs, expectedIDs) {
                    t.Errorf("ordering mismatch: got %v, want %v", gotIDs, expectedIDs)
                }
            })
        }
