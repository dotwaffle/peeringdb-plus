<objective>
Fix BUG-02: `poc.role` carries `field.String("role").NotEmpty()` which would
abort incremental sync on the FIRST poc tombstone arriving from upstream
PeeringDB's `?since=` window. Empirical confirmation 2026-04-26 (live spike
in SEED-001 / `260426-pms-SUMMARY.md` § Out of Scope Findings):
upstream emits `status="deleted"` rows with PII-scrubbed empty strings on
multiple text fields. The 6 folded entities had `name=""` already verified
on the spike (5 of 101 `poc` rows in the 30-day window were tombstoned);
`role` is the next obvious PII-scrub target on the same `poc` rows that
already shipped tombstones to the spike, and the `NotEmpty()` validator
would reject them at the upsert builder.

Per CONTEXT.md D-02, the fix scope is wide: audit ALL 13 entities' string
fields for `NotEmpty()`, decide drop-vs-keep per tombstone-vulnerability,
and apply via `cmd/pdb-schema-generate` (codegen-layer per the 260426-pms
pattern). Direct edits to the regenerated schemas would be silently stripped
on the next `go generate ./...`.

Per CONTEXT.md D-03, both a unit test (fast feedback during development)
AND an httptest fake-upstream conformance test (regression guard mirroring
260426-pms `TestSync_IncrementalDeletionTombstone`) are required.

Purpose: prevent the next "incremental sync silently aborted on first poc
tombstone" production incident before it happens. The 260426-pms quick
task already shipped the fix for `name`; this is the symmetric fix for
`role`. Out-of-scope per CONTEXT.md: `ixprefix.prefix` (structural — IP
prefixes are not in the GDPR PII-scrub set; tombstones for ixprefix retain
their prefix value because the prefix is the row's identity).

Output: extended `cmd/pdb-schema-generate` predicate (`isTombstoneVulnerableField`)
covering both `name` and `role`; regenerated `ent/schema/poc.go` without
`.NotEmpty()` on role; regenerated ent client artefacts (validator
declarations, check() blocks, runtime initialiser) no longer carrying
`RoleValidator`; new `TestSync_IncrementalRoleTombstone` end-to-end
regression guard; extended `cmd/pdb-schema-generate/main_test.go`
`notWantParts` regression guards; ixprefix.prefix UNTOUCHED.

- `name=""` (PII-scrubbed)
- other PII fields scrubbed similarly

The spike DID NOT explicitly inspect the `role` field on those 5 tombstones —
the 260426-pms scope was strict. However, the GDPR-style scrub pattern
applies uniformly: any field carrying user identity-correlated data
(name, role, email, phone) is a candidate. `role` (e.g. "Technical",
"Policy", "NOC") may or may not be scrubbed by upstream — but the
`NotEmpty()` validator means we cannot safely tolerate an empty value
even if upstream sometimes does emit `role=""`. Belt-and-braces.

The Phase 73 fix is preventive — drop the validator BEFORE the next
poc tombstone arrives in production with `role=""`.

// guard: upstream PeeringDB ?since= responses can carry status='deleted'
// poc tombstones with PII-scrubbed role="". This drives a tombstone for
// an existing poc through an incremental sync cycle and asserts the row
// is soft-deleted (not hard-deleted) and excluded from the anonymous
// (status="ok") list path.
//
// The empty-role path is the load-bearing assertion: if the NotEmpty()
// validator removed from poc.role in Phase 73 ever re-appears, this test
// fails on the second sync's upsert with "validator failed for field
// Poc.role".
//
// Mirrors TestSync_IncrementalDeletionTombstone (260426-pms) structure.
func TestSync_IncrementalRoleTombstone(t *testing.T) {
    t.Parallel()
    t1 := float64(time.Date(2026, 4, 26, 12, 0, 0, 0, time.UTC).Unix())
    f := newFixtureWithMeta(t, t1)
    // Parent network for the pocs.
    f.responses["net"] = []any{makeNet(50, 1, 64500, "ParentNet", "ok")}
    // (Org parent for the network — fixtureWithMeta default emits empty.
    // The sync worker tolerates a missing org parent; if the test fails
    // because the network upsert chokes on no-such-org_id, seed the org
    // explicitly: f.responses["org"] = []any{makeOrg(1, "Org", "ok")}.)
    f.responses["org"] = []any{makeOrg(1, "Org1", "ok")}
    f.responses["poc"] = []any{
        makePoc(10, 50, "Alice", "Technical", "ok"),
        makePoc(11, 50, "Bob", "Policy", "ok"),
    }

    w, db := newTestWorkerWithMode(t, f.server.URL, config.SyncModeIncremental)
    ctx := t.Context()

    // Cycle 1: cursor zero → falls back to full sync.
    if err := w.Sync(ctx, config.SyncModeIncremental); err != nil {
        t.Fatalf("first sync: %v", err)
    }
    pocCount, err := w.entClient.Poc.Query().Count(ctx)
    if err != nil {
        t.Fatalf("count pocs after cycle 1: %v", err)
    }
    if pocCount != 2 {
        t.Fatalf("expected 2 pocs after cycle 1, got %d", pocCount)
    }
    cursor, err := GetCursor(ctx, db, "poc")
    if err != nil {
        t.Fatalf("get cursor after cycle 1: %v", err)
    }
    if cursor.IsZero() {
        t.Fatal("expected non-zero cursor after cycle 1")
    }

    // Reset since-tracking and bump generated for cycle 2.
    for _, v := range f.sinceSeen {
        v.Store(false)
    }
    t2 := t1 + 3600
    f.generated = t2

    // Cycle 2: upstream returns Alice unchanged + Bob tombstoned with
    // role="" (PII-scrub style).
    f.responses["poc"] = []any{
        makePoc(10, 50, "Alice", "Technical", "ok"),
        makePoc(11, 50, "", "", "deleted"),
    }
    if err := w.Sync(ctx, config.SyncModeIncremental); err != nil {
        t.Fatalf("second sync: %v", err)
    }

    // (a) ?since= sent on cycle 2.
    if seen, ok := f.sinceSeen["poc"]; !ok || !seen.Load() {
        t.Error("expected ?since= for poc on cycle 2 (incremental cursor present)")
    }

    // (b) Soft-delete: row preserved, role scrubbed, status flipped.
    totalCount, err := w.entClient.Poc.Query().Count(ctx)
    if err != nil {
        t.Fatalf("count pocs after cycle 2: %v", err)
    }
    if totalCount != 2 {
        t.Fatalf("expected 2 pocs after cycle 2, got %d", totalCount)
    }

    alice, err := w.entClient.Poc.Get(ctx, 10)
    if err != nil {
        t.Fatalf("get poc 10: %v", err)
    }
    if alice.Status != "ok" || alice.Role != "Technical" {
        t.Errorf("alice = {status=%q, role=%q}, want {status=\"ok\", role=\"Technical\"}", alice.Status, alice.Role)
    }

    bob, err := w.entClient.Poc.Get(ctx, 11)
    if err != nil {
        t.Fatalf("get poc 11: %v", err)
    }
    if bob.Status != "deleted" {
        t.Errorf("bob status = %q, want %q", bob.Status, "deleted")
    }
    if bob.Role != "" {
        t.Errorf("bob role = %q, want %q (PII-scrubbed tombstone)", bob.Role, "")
    }

    // (c) Anonymous list path: only the live poc.
    live, err := w.entClient.Poc.Query().Where(poc.StatusEQ("ok")).All(ctx)
    if err != nil {
        t.Fatalf("query live pocs: %v", err)
    }
    if len(live) != 1 {
        t.Fatalf("expected 1 live poc, got %d", len(live))
    }
    if live[0].ID != 10 {
        t.Errorf("live poc ID = %d, want 10", live[0].ID)
    }
}
