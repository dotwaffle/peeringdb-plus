D-15 correctness contract: seed a `visible="Users"` POC via the bypass, then
issue anonymous HTTP requests through the FULL real middleware chain to EVERY
read surface, and assert the row is absent. Flip `PDBPLUS_PUBLIC_TIER=users`,
re-run, and assert the row is present.

This is the single integration test that proves Phases 59-01 through 59-05
compose correctly: config → middleware → privctx → ent policy → per-surface
handler. If any link is wrong (tier not stamped, policy not wired, bypass
leaked into handler, surface-native 404 not returned), this test fails.

Per D-14, every surface returns a "not found" idiom rather than 403 — this
prevents the existence-leak attack where a caller maps private rows by
distinguishing "filtered" from "missing".

Output: One integration test file exercising the full stack across all
5 surfaces and both tier values.

    func TestE2E_AnonymousCannotSeeUsersPoc(t *testing.T) {
        t.Parallel()
        fix := buildE2EFixture(t, privctx.TierPublic)

        // Surface 1: pdbcompat /api (detail + list)
        t.Run("pdbcompat_detail", func(t *testing.T) {
            resp, err := http.Get(fix.server.URL + "/api/poc/" + strconv.Itoa(fix.pocID))
            requireNoErr(t, err)
            defer resp.Body.Close()
            if resp.StatusCode != http.StatusNotFound {
                t.Fatalf("expected 404, got %d", resp.StatusCode)
            }
        })
        t.Run("pdbcompat_list", func(t *testing.T) {
            resp, err := http.Get(fix.server.URL + "/api/poc")
            requireNoErr(t, err)
            defer resp.Body.Close()
            var body struct{ Data []map[string]any `json:"data"` }
            if err := json.NewDecoder(resp.Body).Decode(&body); err != nil { t.Fatal(err) }
            for _, row := range body.Data {
                if int(row["id"].(float64)) == fix.pocID {
                    t.Fatalf("Users POC leaked into /api/poc list")
                }
            }
        })

        // Surface 2: entrest /rest/v1/pocs (detail + list)
        t.Run("rest_detail", func(t *testing.T) {
            resp, err := http.Get(fix.server.URL + "/rest/v1/pocs/" + strconv.Itoa(fix.pocID))
            requireNoErr(t, err)
            defer resp.Body.Close()
            if resp.StatusCode != http.StatusNotFound {
                t.Fatalf("expected 404, got %d", resp.StatusCode)
            }
        })
        t.Run("rest_list", func(t *testing.T) { /* similar: fetch /rest/v1/pocs, assert pocID absent */ })

        // Surface 3: ConnectRPC PocService.Get + List
        t.Run("grpc_get", func(t *testing.T) {
            cl := peeringdbv1connect.NewPocServiceClient(http.DefaultClient, fix.server.URL)
            _, err := cl.Get(context.Background(), connect.NewRequest(&peeringdbv1.GetRequest{Id: int64(fix.pocID)}))
            if err == nil {
                t.Fatalf("expected CodeNotFound, got nil error")
            }
            if ce, ok := err.(*connect.Error); !ok || ce.Code() != connect.CodeNotFound {
                t.Fatalf("expected CodeNotFound, got %v", err)
            }
        })
        t.Run("grpc_list", func(t *testing.T) { /* similar: List, assert pocID absent from returned slice */ })

        // Surface 4: GraphQL — node + list
        t.Run("graphql_node", func(t *testing.T) {
            body := fmt.Sprintf(`{"query":"query($id:ID!){poc(id:$id){id}}","variables":{"id":"%d"}}`, fix.pocID)
            resp, err := http.Post(fix.server.URL + "/graphql", "application/json", strings.NewReader(body))
            requireNoErr(t, err)
            defer resp.Body.Close()
            var gr struct{
                Data   struct{ Poc *struct{ Id string } `json:"poc"` } `json:"data"`
                Errors []struct{ Message string; Extensions map[string]any } `json:"errors"`
            }
            _ = json.NewDecoder(resp.Body).Decode(&gr)
            if gr.Data.Poc != nil {
                t.Fatalf("expected poc=null, got %+v", gr.Data.Poc)
            }
            // GraphQL may return null+no-error (ent.IsNotFound classified as soft miss) OR
            // null+error with NOT_FOUND code. Accept both per D-13.
        })
        t.Run("graphql_list", func(t *testing.T) { /* query pocs{edges{node{id}}}, assert pocID absent */ })

        // Surface 5: /ui/ detail
        t.Run("ui_detail", func(t *testing.T) {
            req, _ := http.NewRequest("GET", fix.server.URL + "/ui/poc/" + strconv.Itoa(fix.pocID), nil)
            req.Header.Set("User-Agent", "Mozilla/5.0")  // force HTML per CLAUDE.md content negotiation note
            resp, err := http.DefaultClient.Do(req)
            requireNoErr(t, err)
            defer resp.Body.Close()
            if resp.StatusCode != http.StatusNotFound {
                t.Fatalf("expected 404 on /ui/poc/%d, got %d", fix.pocID, resp.StatusCode)
            }
        })
    }

      the default global tracer provider is a no-op under test, so no special
      handling needed.

    - **gRPC client construction:** use `connectrpc.com/connect` stdlib client

      ConnectRPC handlers are standard http.Handlers mounted into the real mux.

    - **GraphQL query format:** POST JSON to `/graphql`. Use the project's
