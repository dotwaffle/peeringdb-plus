```go
func TestFixtures_StatusSanity(t *testing.T) {
    t.Parallel()
    if len(parity.StatusFixtures) == 0 { t.Fatal("StatusFixtures empty") }
    statuses := map[string]int{}
    hasCampusPending := false
    for _, f := range parity.StatusFixtures {
        if f.Entity == "" || f.Upstream == "" { t.Errorf("%+v", f) }
        if s, ok := f.Fields["status"].(string); ok {
            statuses[s]++
            if f.Entity == "campus" && s == "pending" { hasCampusPending = true }
        }
    }
    if len(statuses) < 3 { t.Errorf("want ≥3 distinct statuses; got %v", statuses) }
    if !hasCampusPending { t.Error("STATUS-03 carve-out: no (campus, pending) fixture") }
}

func TestFixtures_LimitSanity(t *testing.T) {
    t.Parallel()
    if len(parity.LimitFixtures) == 0 { t.Fatal("LimitFixtures empty") }
    var networkCount int
    for _, f := range parity.LimitFixtures {
        if f.Entity == "network" { networkCount++ }
    }
    if networkCount < 260 {
        t.Errorf("LIMIT-01 unlimited boundary: want ≥260 network fixtures; got %d", networkCount)
    }
}
```
