First extend `parseFieldOp` per D-06:

```go
// parseFieldOp splits a filter parameter key into relation segments,
// final field name, and operator. Syntax:
//
//   [<relation>__]* <field> [__<op>]
//
// Returns up to 2 relation segments (D-04 hard cap enforced at caller);
// a len(relationSegments) > 2 return value is a signal to the caller
// that the key is too deep to traverse — caller MUST treat as unknown
// per D-04/D-05.
//
// Empty field name or malformed input (leading "__", consecutive "__"
// producing empty segments) returns the best-effort split; callers
// validate finalField != "" and reject otherwise.
func parseFieldOp(key string) (relationSegments []string, finalField string, op string) {
    parts := strings.Split(key, "__")
    // Last segment might be an operator OR the final field name.
    // Walk backwards: if the last segment is a known operator, strip it.
    op = ""
    if len(parts) >= 2 && isKnownOperator(parts[len(parts)-1]) {
        op = parts[len(parts)-1]
        parts = parts[:len(parts)-1]
    }
    if len(parts) == 0 {
        return nil, "", op
    }
    finalField = parts[len(parts)-1]
    if len(parts) == 1 {
        return nil, finalField, op
    }
    relationSegments = parts[:len(parts)-1]
    return relationSegments, finalField, op
}

// isKnownOperator reports whether suffix matches an operator supported
// by buildPredicate (exact/contains/startswith/in/lt/gt/lte/gte and
// their case-insensitive siblings). Used to disambiguate
// "field__op" from "field_with_double_underscore".
func isKnownOperator(suffix string) bool {
    switch suffix {
    case "contains", "icontains", "startswith", "istartswith", "iexact",
        "in", "lt", "gt", "lte", "gte":
        return true
    }
    return false
}
```

// buildLocalPredicate is the extraction of the pre-Phase-70 behaviour.
// Returns (nil, false, nil) when the field is unknown — caller treats
// that as silent-ignore.
func buildLocalPredicate(field, op, value string, tc TypeConfig) (func(*sql.Selector), bool, error) {
    ft, ok := tc.Fields[field]
    if !ok {
        return nil, false, nil
    }
    folded := tc.FoldedFields[field]
    p, err := buildPredicate(field, op, value, ft, folded)
    if err != nil {
        if errors.Is(err, errEmptyIn) {
            return nil, true, nil
        }
        return nil, false, err
    }
    return p, false, nil
}

// buildTraversalPredicate resolves a 1-hop or 2-hop cross-entity filter.
// relSegs has len 1 or 2 (caller enforced D-04 cap).
//
// Resolution order:
//   1. Path A: is the full key in Allowlists[<tc.Name>].Direct (for 1-hop)
//      or Via[<relSegs[0]>] (for 2-hop)?
//   2. Path B: does LookupEdge(tc.Name, relSegs[0]) return a non-excluded
//      edge? If 2-hop, repeat for the target entity.
//
// Returns (predicate, true, false, nil) on success. ok=false => unknown.
// emptyResult=true passes up the empty-__in sentinel.
func buildTraversalPredicate(tc TypeConfig, relSegs []string, field, op, value string) (predicate func(*sql.Selector), ok bool, emptyResult bool, err error) {
    // Reconstruct the full key (for Path A lookup against Allowlists).
    // AllowlistEntry stores keys WITHOUT the operator suffix, so strip
    // op-stripped form.
    fullKey := strings.Join(relSegs, "__") + "__" + field

    // Path A.
    if entry, hasAllowlist := Allowlists[tc.Name]; hasAllowlist {
        if len(relSegs) == 1 {
            if slices.Contains(entry.Direct, fullKey) {
                return buildSinglHop(tc.Name, relSegs[0], field, op, value)
            }
        } else {
            // 2-hop
            if tails, okVia := entry.Via[relSegs[0]]; okVia {
                tailKey := relSegs[1] + "__" + field
                if slices.Contains(tails, tailKey) {
                    return buildTwoHop(tc.Name, relSegs[0], relSegs[1], field, op, value)
                }
            }
        }
    }

    // Path B: try introspection.
    edge, okEdge := LookupEdge(tc.Name, relSegs[0])
    if !okEdge {
        return nil, false, false, nil
    }
    if len(relSegs) == 1 {
        // Does the target type have this field?
        if _, hasField := TargetFields(edge.TargetType)[field]; !hasField {
            return nil, false, false, nil
        }
        return buildSinglHop(tc.Name, relSegs[0], field, op, value)
    }
    // 2-hop Path B: second hop must also exist.
    edge2, okEdge2 := LookupEdge(edge.TargetType, relSegs[1])
    if !okEdge2 {
        return nil, false, false, nil
    }
    if _, hasField := TargetFields(edge2.TargetType)[field]; !hasField {
        return nil, false, false, nil
    }
    return buildTwoHop(tc.Name, relSegs[0], relSegs[1], field, op, value)
}

func buildSinglHop(entityType, fk, field, op, value string) (func(*sql.Selector), bool, bool, error) {
    edge, ok := LookupEdge(entityType, fk)
    if !ok {
        return nil, false, false, nil
    }
    targetFt, ok := TargetFields(edge.TargetType)[field]
    if !ok {
        return nil, false, false, nil
    }
    // Build inner predicate on the target table's field.
    targetFolded := Registry[edge.TargetType].FoldedFields[field]
    innerPred, err := buildPredicate(field, op, value, targetFt, targetFolded)
    if err != nil {
        if errors.Is(err, errEmptyIn) {
            return nil, false, true, nil
        }
        return nil, false, false, err
    }
    return func(s *sql.Selector) {
        t := sql.Table(edge.TargetTable)
        sub := sql.Select(t.C(edge.TargetIDColumn)).From(t)
        // Apply innerPred on a selector scoped to the sub-table.
        // sql.Selector is pass-by-pointer; innerPred mutates via .Where.
        // We need to re-target the inner predicate's column references
        // to the sub-table. Since buildPredicate uses sql.FieldX helpers
        // that reference by bare column name (no table prefix), the
        // sub-selector's C() calls still resolve correctly within the
        // subquery's scope.
        innerSel := sql.Select(t.C(edge.TargetIDColumn)).From(t)
        innerPred(innerSel)
        s.Where(sql.In(s.C(edge.ParentFKColumn), innerSel))
        _ = sub // keep lint happy if sub unused in the final shape
    }, true, false, nil
}
