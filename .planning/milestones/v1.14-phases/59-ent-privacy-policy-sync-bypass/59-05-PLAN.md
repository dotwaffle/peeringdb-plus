its context with `privacy.DecisionContext(ctx, privacy.Allow)` at the very top,
BEFORE any ent call, any atomic guard, or any OTel span. Every downstream
upsert, read, and child goroutine inherits the bypass via context.

audit test enforces "exactly one call site" — if a future developer adds a
second `privacy.DecisionContext(*, privacy.Allow)` anywhere, CI fails. That's
the mitigation for T-59-02 (bypass leak into handler goroutine).

Output: One-line change to worker.go + two bypass-behaviour tests + one
source-scanning audit test.

    ```go
    // Source-scanning audit test per D-09 + RESEARCH Pitfall 4.
    //
    // The ent privacy bypass `privacy.DecisionContext(ctx, privacy.Allow)`
    // must appear in exactly ONE production source location:
    // internal/sync/worker.go. A second call site in any non-test file
    // would be a silent policy bypass — this test fails the build.
    //
    // Test files are exempt: they legitimately seed Users-tier rows via
    // the bypass so they can then assert the policy filters correctly.
    package sync

    import (
        "io/fs"
        "os"
        "path/filepath"
        "regexp"
        "strings"
        "testing"
    )

    var bypassCallRE = regexp.MustCompile(`privacy\.DecisionContext\([^)]*privacy\.Allow[^)]*\)`)

    func TestSyncBypass_SingleCallSite(t *testing.T) {
        // Discover repo root by walking up from this test file looking for go.mod.
        root, err := findRepoRoot()
        if err != nil {
            t.Fatalf("finding repo root: %v", err)
        }

        type hit struct{ path string; line int; text string }
        var hits []hit

        scanDirs := []string{"internal", "cmd", "ent/schema"}
        for _, d := range scanDirs {
            base := filepath.Join(root, d)
            err := filepath.WalkDir(base, func(path string, ent fs.DirEntry, err error) error {
                if err != nil { return err }
                if ent.IsDir() { return nil }
                if !strings.HasSuffix(path, ".go") { return nil }
                if strings.HasSuffix(path, "_test.go") { return nil }
                // Skip generated ent/ subtree except ent/schema/ (handwritten).
                if strings.Contains(path, "/ent/") && !strings.Contains(path, "/ent/schema/") {
                    return nil
                }
                b, err := os.ReadFile(path)
                if err != nil { return err }
                for i, line := range strings.Split(string(b), "\n") {
                    if bypassCallRE.MatchString(line) {
                        hits = append(hits, hit{path: path, line: i + 1, text: strings.TrimSpace(line)})
                    }
                }
                return nil
            })
            if err != nil {
                t.Fatalf("walking %s: %v", base, err)
            }
        }

        // Multi-line matches: fallback scan treating file as single string
        // (regex above uses [^)]* so it won't cross newlines; do a second
        // pass for the multi-line construction)
        for _, d := range scanDirs {
            base := filepath.Join(root, d)
            err := filepath.WalkDir(base, func(path string, ent fs.DirEntry, err error) error {
                if err != nil { return err }
                if ent.IsDir() || !strings.HasSuffix(path, ".go") { return nil }
                if strings.HasSuffix(path, "_test.go") { return nil }
                if strings.Contains(path, "/ent/") && !strings.Contains(path, "/ent/schema/") { return nil }
                b, err := os.ReadFile(path)
                if err != nil { return err }
                content := string(b)
                // Collapse whitespace for multi-line DecisionContext(ctx,\n privacy.Allow) calls.
                collapsed := regexp.MustCompile(`\s+`).ReplaceAllString(content, " ")
                for _, m := range regexp.MustCompile(`privacy\.DecisionContext\([^)]*privacy\.Allow[^)]*\)`).FindAllStringIndex(collapsed, -1) {
                    // Find line number in original by counting newlines up to m[0] — approximation.
                    // But we already captured single-line matches above; only add if not seen.
                    _ = m
                    // For simplicity: if any multi-line match appears, count per file — the single-line
                    // walker above suffices in practice since gofmt collapses DecisionContext to one line.
                }
                return nil
            })
            if err != nil { t.Fatalf("walking %s: %v", base, err) }
        }

        const wantPath = "internal/sync/worker.go"
        if len(hits) == 0 {
            t.Fatalf("expected exactly 1 call to privacy.DecisionContext(ctx, privacy.Allow) in production code; found 0 (did Plan 05 Task 1 run?)")
        }
        if len(hits) > 1 {
            var msg strings.Builder
            msg.WriteString("expected exactly 1 bypass call site per D-09; found:\n")
            for _, h := range hits {
                msg.WriteString("  ")
                msg.WriteString(h.path)
                msg.WriteString(":")
                msg.WriteString(itoa(h.line))
                msg.WriteString("  ")
                msg.WriteString(h.text)
                msg.WriteString("\n")
            }
            t.Fatal(msg.String())
        }
        if !strings.HasSuffix(hits[0].path, wantPath) {
            t.Fatalf("bypass call must be in %s; found in %s:%d", wantPath, hits[0].path, hits[0].line)
        }
    }

    func findRepoRoot() (string, error) {
        dir, err := os.Getwd()
        if err != nil { return "", err }
        for {
            if _, err := os.Stat(filepath.Join(dir, "go.mod")); err == nil { return dir, nil }
            parent := filepath.Dir(dir)
            if parent == dir { return "", os.ErrNotExist }
            dir = parent
        }
    }

    func itoa(i int) string { /* use strconv.Itoa — or just fmt.Sprintf */ }
    ```

    Polish: use `strconv.Itoa` instead of the stub `itoa`, and delete the
    multi-line fallback block (the single-line scan is sufficient because
    `gofmt` keeps `privacy.DecisionContext(ctx, privacy.Allow)` on one line

    The test is deliberately source-scanning (not AST-based) — it's cheap,
    robust to ent codegen changes, and matches the "grep-able audit" phrasing
    in D-09 and RESEARCH §"Pitfall 4".
