package visbaseline

import (
    "errors"
    "fmt"
    "io/fs"
    "os"
    "path/filepath"
)

// RedactDir walks srcAuthDir (the /tmp raw-auth tree produced by Capture.Run)
// and, for each .json file found, reads the matching anon file from anonDir
// and writes the redacted output to dstAuthDir, preserving directory layout.
//
// Expected layout in src dirs: <root>/<mode>/api/<type>/page-<n>.json.
// srcAuthDir should be the "auth" side; anonDir should be the "anon" side.
// Example: capture.Run() returns rawAuthDir = /tmp/pdb-vis-capture-XXXX;
// the actual auth tree lives at rawAuthDir/auth/api/<type>/page-<n>.json.
// Callers should pass the FULL srcAuthDir = rawAuthDir + "/auth" and
// anonDir = <repo-out>/anon (the committed anon tree).
//
// RedactDir fails HARD if any auth file lacks a matching anon pair. It does
// NOT silently write an all-placeholder output because such a file would be
// indistinguishable from "this type has zero overlap" â€” ambiguous and unsafe.
func RedactDir(srcAuthDir, anonDir, dstAuthDir string) error {
    if !dirExists(srcAuthDir) {
        return fmt.Errorf("RedactDir: srcAuthDir not found: %s", srcAuthDir)
    }
    if !dirExists(anonDir) {
        return fmt.Errorf("RedactDir: anonDir not found: %s", anonDir)
    }
    if err := os.MkdirAll(dstAuthDir, 0o700); err != nil {
        return fmt.Errorf("RedactDir: create dst %s: %w", dstAuthDir, err)
    }

    return filepath.WalkDir(srcAuthDir, func(path string, d fs.DirEntry, walkErr error) error {
        if walkErr != nil {
            return walkErr
        }
        if d.IsDir() || filepath.Ext(path) != ".json" {
            return nil
        }

        rel, err := filepath.Rel(srcAuthDir, path)
        if err != nil {
            return fmt.Errorf("compute rel path for %s: %w", path, err)
        }
        // rel looks like "api/<type>/page-<n>.json" (we stripped the "auth"
        // component because callers passed srcAuthDir = .../auth).
        anonPath := filepath.Join(anonDir, rel)
        if _, err := os.Stat(anonPath); err != nil {
            if errors.Is(err, os.ErrNotExist) {
                return fmt.Errorf("no matching anon fixture for %s (expected at %s)", path, anonPath)
            }
            return fmt.Errorf("stat %s: %w", anonPath, err)
        }

        authBytes, err := os.ReadFile(path)
        if err != nil {
            return fmt.Errorf("read auth %s: %w", path, err)
        }
        anonBytes, err := os.ReadFile(anonPath)
        if err != nil {
            return fmt.Errorf("read anon %s: %w", anonPath, err)
        }

        redacted, err := Redact(anonBytes, authBytes)
        if err != nil {
            return fmt.Errorf("redact %s: %w", path, err)
        }

        dstPath := filepath.Join(dstAuthDir, rel)
        if err := os.MkdirAll(filepath.Dir(dstPath), 0o700); err != nil {
            return fmt.Errorf("mkdir %s: %w", filepath.Dir(dstPath), err)
        }
        if err := os.WriteFile(dstPath, redacted, 0o600); err != nil {
            return fmt.Errorf("write redacted %s: %w", dstPath, err)
        }
        return nil
    })
}

func dirExists(path string) bool {
    info, err := os.Stat(path)
    return err == nil && info.IsDir()
}

package visbaseline

import (
    "fmt"
    "os"
    "path/filepath"
    "sort"
    "time"
)

// BuildReport walks an anon tree + a redacted-auth tree and constructs a
// Report covering every PeeringDB type present in both. Pages are aggregated:
// row counts for page-1 and page-2 are SUMMED into the type total.
//
// Expected layout in each root: <root>/api/<type>/page-<n>.json (callers
// should pass the per-target per-mode subdirs, e.g.
// testdata/visibility-baseline/beta/anon and testdata/visibility-baseline/beta/auth).
func BuildReport(anonRoot, authRoot string, targets []string, now time.Time) (Report, error) {
    if !dirExists(anonRoot) {
        return Report{}, fmt.Errorf("BuildReport: anonRoot not found: %s", anonRoot)
    }
    if !dirExists(authRoot) {
        return Report{}, fmt.Errorf("BuildReport: authRoot not found: %s", authRoot)
    }

    // Discover types by listing anonRoot/api/.
    apiDir := filepath.Join(anonRoot, "api")
    entries, err := os.ReadDir(apiDir)
    if err != nil {
        return Report{}, fmt.Errorf("read %s: %w", apiDir, err)
    }

    rep := Report{
        SchemaVersion: ReportSchemaVersion,
        GeneratedAt:   now,
        Targets:       append([]string{}, targets...),
        Types:         make(map[string]TypeReport),
    }

    typeNames := make([]string, 0, len(entries))
    for _, e := range entries {
        if e.IsDir() {
            typeNames = append(typeNames, e.Name())
        }
    }
    sort.Strings(typeNames)

    for _, typeName := range typeNames {
        tr, err := diffTypeAcrossPages(typeName, anonRoot, authRoot)
        if err != nil {
            return Report{}, fmt.Errorf("diff %s: %w", typeName, err)
        }
        rep.Types[typeName] = tr
    }
    return rep, nil
}

// diffTypeAcrossPages sums row counts and field deltas across all pages for
// one type. Missing pages are skipped without error.
func diffTypeAcrossPages(typeName, anonRoot, authRoot string) (TypeReport, error) {
    var agg TypeReport
    seenFields := make(map[string]FieldDelta)

    for page := 1; page <= 10; page++ { // generous upper bound; real use is 2
        anonPath := filepath.Join(anonRoot, "api", typeName, fmt.Sprintf("page-%d.json", page))
        authPath := filepath.Join(authRoot, "api", typeName, fmt.Sprintf("page-%d.json", page))
        anonBytes, err := os.ReadFile(anonPath)
        if err != nil {
            // missing page: stop walking pages for this type
            break
        }
        authBytes, err := os.ReadFile(authPath)
        if err != nil {
            // auth for this page is missing â€” this is NOT fatal; we may have
            // an anon-only prod run. Continue with anon-only accounting.
            authBytes = []byte(`{"meta":{},"data":[]}`)
        }

        pt, err := Diff(typeName, anonBytes, authBytes)
        if err != nil {
            return TypeReport{}, fmt.Errorf("%s page %d: %w", typeName, page, err)
        }

        agg.AnonRowCount += pt.AnonRowCount
        agg.AuthRowCount += pt.AuthRowCount
        agg.AuthOnlyRowCount += pt.AuthOnlyRowCount
        agg.VisibleValuesAnon = mergeSets(agg.VisibleValuesAnon, pt.VisibleValuesAnon)
        agg.VisibleValuesAuth = mergeSets(agg.VisibleValuesAuth, pt.VisibleValuesAuth)

        for _, fd := range pt.Fields {
            existing, ok := seenFields[fd.Name]
            if !ok {
                seenFields[fd.Name] = fd
                continue
            }
            existing.RowsAdded += fd.RowsAdded
            existing.AuthOnly = existing.AuthOnly || fd.AuthOnly
            existing.ValueSetDrift = existing.ValueSetDrift || fd.ValueSetDrift
            seenFields[fd.Name] = existing
        }
    }

    names := make([]string, 0, len(seenFields))
    for n := range seenFields {
        names = append(names, n)
    }
    sort.Strings(names)
    for _, n := range names {
        agg.Fields = append(agg.Fields, seenFields[n])
    }
    return agg, nil
}

func mergeSets(a, b []string) []string {
    set := make(map[string]struct{}, len(a)+len(b))
    for _, v := range a {
        set[v] = struct{}{}
    }
    for _, v := range b {
        set[v] = struct{}{}
    }
    out := make([]string, 0, len(set))
    for v := range set {
        out = append(out, v)
    }
    sort.Strings(out)
    return out
}

func runRedact(cfg runConfig, logger *slog.Logger) error {
    if cfg.srcDir == "" {
        return errors.New("-redact requires -src-auth pointing to a raw-auth tree")
    }
    outDir := cfg.outDir
    if outDir == "" {
        return errors.New("-redact requires -out pointing to the anon-tree root (redactor needs anon pairs)")
    }
    anonDir := filepath.Join(outDir, "anon")
    dstAuthDir := cfg.dstDir
    if dstAuthDir == "" {
        dstAuthDir = filepath.Join(outDir, "auth")
    }
    logger.LogAttrs(context.Background(), slog.LevelInfo, "redacting auth tree",
        slog.String("src_auth", cfg.srcDir),
        slog.String("anon", anonDir),
        slog.String("dst_auth", dstAuthDir),
    )
    if err := visbaseline.RedactDir(cfg.srcDir, anonDir, dstAuthDir); err != nil {
        return fmt.Errorf("redact: %w", err)
    }
    fmt.Fprintf(os.Stdout, "Redaction complete: %s\n", dstAuthDir)
    return nil
}

// runDiff walks the baseline root (e.g. testdata/visibility-baseline/) and
// emits one DIFF.md + one diff.json covering EVERY {target}/ subdir it finds
// that contains the anon/ + auth/ pair. This replaces the earlier single-
// target behaviour so running the command twice does not clobber the output.
//
// -out semantics:
//   If -out points at a directory that itself contains anon/ and auth/
//   subdirs (legacy: testdata/visibility-baseline/beta), the function treats
//   it as ONE target named filepath.Base(outDir). The DIFF.md + diff.json
//   land in the parent (testdata/visibility-baseline/).
//   If -out points at the baseline root (testdata/visibility-baseline/), the
//   function auto-discovers subdirs beta/, prod/, etc. The DIFF.md + diff.json
//   land at -out itself.
//
// GO-CFG-1 fail-fast: the function validates -out on entry and refuses
// - empty -out
// - -out where filepath.Dir(outDir) == outDir (i.e. root "/" or ".")
// - single-target mode where neither anon/ nor auth/ subdir exists
// - baseline-root mode where no {target}/anon or {target}/auth subdir exists
func runDiff(cfg runConfig, logger *slog.Logger) error {
    outDir := cfg.outDir
    if outDir == "" {
        return errors.New("-diff requires -out pointing to a target dir (beta/prod) or the baseline root")
    }
    absOut, err := filepath.Abs(outDir)
    if err != nil {
        return fmt.Errorf("resolve -out: %w", err)
    }
    if filepath.Dir(absOut) == absOut {
        return fmt.Errorf("-out=%q resolves to filesystem root; refusing", outDir)
    }

    // Decide: single-target vs multi-target mode by probing for anon/auth.
    _, anonErr := os.Stat(filepath.Join(outDir, "anon"))
    _, authErr := os.Stat(filepath.Join(outDir, "auth"))
    singleTarget := anonErr == nil && authErr == nil

    type targetRun struct {
        name     string
        anonRoot string
        authRoot string
    }
    var runs []targetRun
    var baselineRoot string

    if singleTarget {
        target := filepath.Base(outDir)
        runs = append(runs, targetRun{
            name:     target,
            anonRoot: filepath.Join(outDir, "anon"),
            authRoot: filepath.Join(outDir, "auth"),
        })
        baselineRoot = filepath.Dir(outDir)
    } else {
        // Multi-target: outDir IS the baseline root. Walk subdirs.
        baselineRoot = outDir
        entries, err := os.ReadDir(outDir)
        if err != nil {
            return fmt.Errorf("read baseline root %s: %w", outDir, err)
        }
        for _, e := range entries {
            if !e.IsDir() {
                continue
            }
            name := e.Name()
            sub := filepath.Join(outDir, name)
            anonSub := filepath.Join(sub, "anon")
            authSub := filepath.Join(sub, "auth")
            if _, err := os.Stat(anonSub); err != nil {
                continue
            }
            runs = append(runs, targetRun{
                name:     name,
                anonRoot: anonSub,
                authRoot: authSub, // may not exist; BuildReport handles anon-only
            })
        }
        if len(runs) == 0 {
            return fmt.Errorf("-out=%q has no {target}/anon/ subdirs; nothing to diff", outDir)
        }
    }

    // Build a combined Report by merging per-target TypeReports under distinct
    // target keys in the JSON schema. For back-compat, also emit per-target
    // DIFF-{target}.md alongside the unified DIFF.md.
    unified := visbaseline.Report{
        SchemaVersion: visbaseline.ReportSchemaVersion,
        GeneratedAt:   time.Now().UTC(),
        Targets:       make([]string, 0, len(runs)),
        Types:         make(map[string]visbaseline.TypeReport),
    }
    perTarget := make(map[string]visbaseline.Report, len(runs))

    for _, run := range runs {
        rep, err := visbaseline.BuildReport(run.anonRoot, run.authRoot, []string{run.name}, unified.GeneratedAt)
        if err != nil {
            return fmt.Errorf("build report for target %s: %w", run.name, err)
        }
        unified.Targets = append(unified.Targets, run.name)
        perTarget[run.name] = rep
        // Namespace type keys by target to avoid collision: "{target}/{type}".
        for typeName, tr := range rep.Types {
            unified.Types[run.name+"/"+typeName] = tr
        }
    }

    // Write unified artifacts at baselineRoot.
    mdPath := filepath.Join(baselineRoot, "DIFF.md")
    jsonPath := filepath.Join(baselineRoot, "diff.json")

    mdFile, err := os.Create(mdPath)
    if err != nil {
        return fmt.Errorf("create %s: %w", mdPath, err)
    }
    defer mdFile.Close()
    if err := visbaseline.WriteMarkdown(mdFile, unified); err != nil {
        return fmt.Errorf("write markdown: %w", err)
    }

    jsonFile, err := os.Create(jsonPath)
    if err != nil {
        return fmt.Errorf("create %s: %w", jsonPath, err)
    }
    defer jsonFile.Close()
    if err := visbaseline.WriteJSON(jsonFile, unified); err != nil {
        return fmt.Errorf("write json: %w", err)
    }

    // Also emit per-target DIFF-{target}.md so reviewers can scope one at a time.
    for _, run := range runs {
        perPath := filepath.Join(baselineRoot, fmt.Sprintf("DIFF-%s.md", run.name))
        perFile, err := os.Create(perPath)
        if err != nil {
            return fmt.Errorf("create %s: %w", perPath, err)
        }
        if err := visbaseline.WriteMarkdown(perFile, perTarget[run.name]); err != nil {
            perFile.Close()
            return fmt.Errorf("write per-target markdown %s: %w", perPath, err)
        }
        perFile.Close()
    }

    fmt.Fprintf(os.Stdout, "Diff written:\n  %s\n  %s\n", mdPath, jsonPath)
    for _, run := range runs {
        fmt.Fprintf(os.Stdout, "  %s/DIFF-%s.md\n", baselineRoot, run.name)
    }
    return nil
}
