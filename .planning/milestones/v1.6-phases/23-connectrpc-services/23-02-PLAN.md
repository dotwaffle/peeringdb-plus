```go
package grpcserver

import (
    "context"
    "fmt"

    "connectrpc.com/connect"
    "entgo.io/ent"                                         // for ent.Asc
    "github.com/dotwaffle/peeringdb-plus/ent"
    "github.com/dotwaffle/peeringdb-plus/ent/{entitypkg}"  // for {entitypkg}.FieldID
    pb "github.com/dotwaffle/peeringdb-plus/gen/peeringdb/v1"
)

// XxxService implements peeringdbv1connect.XxxServiceHandler.
type XxxService struct {
    Client *ent.Client
}

// GetXxx returns a single xxx by ID.
func (s *XxxService) GetXxx(ctx context.Context, req *pb.GetXxxRequest) (*pb.GetXxxResponse, error) {
    entity, err := s.Client.Xxx.Get(ctx, int(req.GetId()))
    if err != nil {
        if ent.IsNotFound(err) {
            return nil, connect.NewError(connect.CodeNotFound,
                fmt.Errorf("entity xxx %d not found", req.GetId()))
        }
        return nil, connect.NewError(connect.CodeInternal,
            fmt.Errorf("get xxx %d: %w", req.GetId(), err))
    }
    return &pb.GetXxxResponse{Xxx: xxxToProto(entity)}, nil
}

// ListXxxs returns a paginated list of xxxs.
func (s *XxxService) ListXxxs(ctx context.Context, req *pb.ListXxxsRequest) (*pb.ListXxxsResponse, error) {
    pageSize := normalizePageSize(req.GetPageSize())
    offset, err := decodePageToken(req.GetPageToken())
    if err != nil {
        return nil, connect.NewError(connect.CodeInvalidArgument,
            fmt.Errorf("invalid page token: %w", err))
    }
    entities, err := s.Client.Xxx.Query().
        Order(ent.Asc(xxx.FieldID)).
        Limit(pageSize + 1).
        Offset(offset).
        All(ctx)
    if err != nil {
        return nil, connect.NewError(connect.CodeInternal,
            fmt.Errorf("list xxxs: %w", err))
    }
    var nextPageToken string
    if len(entities) > pageSize {
        entities = entities[:pageSize]
        nextPageToken = encodePageToken(offset + pageSize)
    }
    result := make([]*pb.Xxx, len(entities))
    for i, e := range entities {
        result[i] = xxxToProto(e)
    }
    return &pb.ListXxxsResponse{Xxxs: result, NextPageToken: nextPageToken}, nil
}

func xxxToProto(e *ent.Xxx) *pb.Xxx { /* field mapping */ }
```
