    ```go
    package pdbcompat

    import (
        "encoding/json"
        "fmt"
        "net/http"
    )

    // ResponseTooLargeType is the RFC 9457 problem-type URI for pre-flight
    // budget-exceeded 413 responses (Phase 71 D-04).
    const ResponseTooLargeType = "https://peeringdb-plus.fly.dev/errors/response-too-large"

    // BudgetExceeded describes a request whose estimated response size
    // exceeds the configured PDBPLUS_RESPONSE_MEMORY_LIMIT. Populated by
    // CheckBudget; consumed by WriteBudgetProblem (413 writer) and
    // structured-log emission in the handler (Plan 04) + OTel telemetry
    // (Plan 05).
    type BudgetExceeded struct {
        MaxRows        int    `json:"max_rows"`
        BudgetBytes    int64  `json:"budget_bytes"`
        EstimatedBytes int64  `json:"-"`
        Count          int    `json:"-"`
        Entity         string `json:"-"`
        Depth          int    `json:"-"`
    }

    // CheckBudget returns (zero, true) if the request fits under budget,
    // (populated, false) otherwise. budgetBytes == 0 disables the check
    // entirely (returns (zero, true) always) — same semantic as
    // PDBPLUS_SYNC_MEMORY_LIMIT=0.
    //
    // The math: perRow = TypicalRowBytes(entity, depth); estimated =
    // count × perRow; over budget iff estimated > budgetBytes.
    // Conservative by construction — typicalRowBytes is already 2× the
    // measured mean (D-03), and count is a precise SELECT COUNT(*) on
    // the already-filtered query.
    func CheckBudget(count int, entity string, depth int, budgetBytes int64) (BudgetExceeded, bool) {
        if budgetBytes <= 0 {
            return BudgetExceeded{}, true
        }
        perRow := TypicalRowBytes(entity, depth)
        if perRow <= 0 {
            // Defensive: TypicalRowBytes returns defaultRowSize for
            // unknown entities, which is >0. This branch is unreachable
            // in practice but guards against a future map mutation that
            // zeroes a value.
            perRow = defaultRowSize
        }
        estimated := int64(count) * int64(perRow)
        if estimated <= budgetBytes {
            return BudgetExceeded{}, true
        }
        maxRows := int(budgetBytes / int64(perRow))
        return BudgetExceeded{
            MaxRows:        maxRows,
            BudgetBytes:    budgetBytes,
            EstimatedBytes: estimated,
            Count:          count,
            Entity:         entity,
            Depth:          depth,
        }, false
    }

    // budgetProblemBody is the exact on-the-wire shape per D-04. We hand-
    // roll the struct instead of reusing httperr.WriteProblem because
    // httperr.ProblemDetail hardcodes Type="about:blank" and lacks the
    // max_rows / budget_bytes extensions. Keeping the custom shape in a
    // local struct avoids leaking the extension fields into the shared
    // httperr types (yagni).
    type budgetProblemBody struct {
        Type        string `json:"type"`
        Title       string `json:"title"`
        Status      int    `json:"status"`
        Detail      string `json:"detail"`
        Instance    string `json:"instance,omitempty"`
        MaxRows     int    `json:"max_rows"`
        BudgetBytes int64  `json:"budget_bytes"`
    }

    // WriteBudgetProblem writes the RFC 9457 413 response described by D-04.
    // Does not set Retry-After — the failure is request-shape, not transient.
    func WriteBudgetProblem(w http.ResponseWriter, instance string, info BudgetExceeded) {
        body := budgetProblemBody{
            Type:   ResponseTooLargeType,
            Title:  "Response exceeds memory budget",
            Status: http.StatusRequestEntityTooLarge,
            Detail: fmt.Sprintf(
                "Request would return ~%d rows totaling ~%d bytes; limit is %d bytes",
                info.Count, info.EstimatedBytes, info.BudgetBytes,
            ),
            Instance:    instance,
            MaxRows:     info.MaxRows,
            BudgetBytes: info.BudgetBytes,
        }
        w.Header().Set("Content-Type", "application/problem+json")
        w.Header().Set("X-Powered-By", poweredByHeader)
        w.WriteHeader(http.StatusRequestEntityTooLarge)
        _ = json.NewEncoder(w).Encode(body)
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| env var → Config | operator-supplied byte size flows into startup validation |
| HTTP request → budget check | count from SELECT COUNT(*) flows into 413 decision |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-71-03-01 | D (Denial of service) | Attacker crafts query with many filters that all return huge counts → repeated 413s consume SELECT COUNT(*) CPU | mitigate | SELECT COUNT(*) on a filtered query is O(rows matching predicates) — bounded by SQLite's index scan cost. Under the Phase 68/69/70 predicates, all filters resolve to indexed columns. A future-phase rate-limit may harden further; for v1.16 the 413 is the user-visible signal. |
| T-71-03-02 | I (Information disclosure) | 413 body reveals total row count to unauthenticated caller | accept | Public read-only mirror — row counts are already observable via the list endpoints (meta.total or len(data)). Exposing the estimated count in 413 is no worse than a successful 200. |
| T-71-03-03 | T (Tampering) | Operator sets PDBPLUS_RESPONSE_MEMORY_LIMIT to an absurdly high value, defeating the guardrail | accept | Config is owner-controlled; documented default sized against replica cap. Misconfig is an operator choice, not an attack surface. |
| T-71-03-04 | E (Elevation of privilege) | Content-Type manipulation in 413 response could confuse intermediate proxies | mitigate | Content-Type: application/problem+json is hardcoded, not user-controlled. Status code 413 hardcoded. |
