```go
// Package testdata provides bench-only fixture helpers.
package testdata

import (
    "context"
    "fmt"
    "math/rand/v2"
    "testing"
    "time"

    "github.com/dotwaffle/peeringdb-plus/ent"
    "github.com/dotwaffle/peeringdb-plus/internal/unifold"
)

// BenchSeedShape captures the row counts for each entity type in the
// traversal bench fixture. Ratios mirror real PeeringDB cardinality
// (approximately): ~10k networks per ~20k facilities per ~1k ixes.
type BenchSeedShape struct {
    Orgs        int
    Networks    int
    Facilities  int
    Ixes        int
    IxLans      int  // 1:1 with Ixes typically
    NetIxLans   int  // ~3-5 per network
}

// Default10k returns the 10k-row bench fixture shape used by
// BenchmarkTraversal_* entries. Total rows ~40k to stress joins while
// fitting in-memory SQLite at <100MB resident.
func Default10k() BenchSeedShape {
    return BenchSeedShape{
        Orgs:       1000,
        Networks:   10000,
        Facilities: 20000,
        Ixes:       1000,
        IxLans:     1000,
        NetIxLans:  30000, // 3 per network
    }
}

// Seed populates client with the shape. Deterministic via fixed RNG
// seed; repeated invocations produce byte-identical row sets (stable
// for benchstat n=6 comparisons).
func Seed(tb testing.TB, client *ent.Client, shape BenchSeedShape) {
    tb.Helper()
    ctx := context.Background()
    r := rand.New(rand.NewPCG(0xCAFEBABE, 0xDEADBEEF))
    now := time.Now()

    // Phase 1: Orgs. Plain Create loop is fine at 1k — ent's bulk
    // create is friendlier but shape stays simple here.
    for i := 1; i <= shape.Orgs; i++ {
        name := fmt.Sprintf("BenchOrg-%06d", i)
        _, err := client.Organization.Create().
            SetID(i).
            SetName(name).
            SetNameFold(unifold.Fold(name)).
            SetStatus("ok").
            SetCreated(now).
            SetUpdated(now).
            Save(ctx)
        if err != nil {
            tb.Fatalf("seed org %d: %v", i, err)
        }
    }

    // Phase 2: Networks in bulk.
    netCreates := make([]*ent.NetworkCreate, 0, shape.Networks)
    for i := 1; i <= shape.Networks; i++ {
        orgID := 1 + r.IntN(shape.Orgs)
        name := fmt.Sprintf("BenchNet-%06d", i)
        netCreates = append(netCreates, client.Network.Create().
            SetID(i).
            SetName(name).
            SetNameFold(unifold.Fold(name)).
            SetAsn(64000+i).
            SetOrgID(orgID).
            SetStatus("ok").
            SetCreated(now).
            SetUpdated(now))
    }
    if _, err := client.Network.CreateBulk(netCreates...).Save(ctx); err != nil {
        tb.Fatalf("bulk seed networks: %v", err)
    }

    // Phase 3: Facilities.
    facCreates := make([]*ent.FacilityCreate, 0, shape.Facilities)
    for i := 1; i <= shape.Facilities; i++ {
        orgID := 1 + r.IntN(shape.Orgs)
        name := fmt.Sprintf("BenchFac-%06d", i)
        facCreates = append(facCreates, client.Facility.Create().
            SetID(i).
            SetName(name).
            SetNameFold(unifold.Fold(name)).
            SetOrgID(orgID).
            SetStatus("ok").
            SetCreated(now).
            SetUpdated(now))
    }
    if _, err := client.Facility.CreateBulk(facCreates...).Save(ctx); err != nil {
        tb.Fatalf("bulk seed facilities: %v", err)
    }

    // Phase 4: Ixes + IxLans (1:1). Each ix belongs to an org.
    for i := 1; i <= shape.Ixes; i++ {
        orgID := 1 + r.IntN(shape.Orgs)
        name := fmt.Sprintf("BenchIX-%06d", i)
        _, err := client.InternetExchange.Create().
            SetID(i).
            SetName(name).
            SetNameFold(unifold.Fold(name)).
            SetOrgID(orgID).
            SetFacCount(r.IntN(20)). // exercises ?...__fac_count__gt=0
            SetStatus("ok").
            SetCreated(now).
            SetUpdated(now).
            Save(ctx)
        if err != nil {
            tb.Fatalf("seed ix %d: %v", i, err)
        }
        _, err = client.IxLan.Create().
            SetID(i).
            SetIxID(i).
            SetStatus("ok").
            SetCreated(now).
            SetUpdated(now).
            Save(ctx)
        if err != nil {
            tb.Fatalf("seed ixlan %d: %v", i, err)
        }
    }

    // Phase 5: NetIxLans — 3 per network, picks a random ixlan.
    nixCreates := make([]*ent.NetworkIxLanCreate, 0, shape.NetIxLans)
    id := 1
    for n := 1; n <= shape.Networks; n++ {
        for j := 0; j < 3; j++ {
            ixlanID := 1 + r.IntN(shape.IxLans)
            nixCreates = append(nixCreates, client.NetworkIxLan.Create().
                SetID(id).
                SetNetID(n).
                SetIxlanID(ixlanID).
                SetIxID(ixlanID).
                SetAsn(64000+n).
                SetStatus("ok").
                SetCreated(now).
                SetUpdated(now))
            id++
        }
    }
    if _, err := client.NetworkIxLan.CreateBulk(nixCreates...).Save(ctx); err != nil {
        tb.Fatalf("bulk seed netixlans: %v", err)
    }
}
```

Notes on size:
- ~1000 + 10000 + 20000 + 1000 + 1000 + 30000 = 63k rows total.
- SQLite in-memory footprint ~25-40 MB.
- Bulk-create batches via `CreateBulk` to keep seed time under ~5s (one-time cost before `b.ResetTimer`).

`math/rand/v2` with explicit `NewPCG(seed1, seed2)` gives deterministic-across-Go-versions behaviour (Go 1.22+). CI-safe.

```go
// TestBenchTraversal_D07_Ceiling is a go-test-time gate that enforces
// D-07's <50ms ceiling on the 2-hop upstream parity case. Runs a single
// warm query (not a bench) and fails the test if wall time exceeds
// 50ms. This catches the worst regressions even in dev workflow
// without relying on CI benchstat.
func TestBenchTraversal_D07_Ceiling(t *testing.T) {
    if testing.Short() {
        t.Skip("skipping 10k-row ceiling check in -short mode")
    }
    h, client := setupBenchHandler(&bencherT{t: t})
    testdata.Seed(&bencherT{t: t}, client, testdata.Default10k())
    req := httptest.NewRequest(http.MethodGet, "/api/fac?ixlan__ix__fac_count__gt=0", nil)
    w := httptest.NewRecorder()
    start := time.Now()
    h.dispatch(w, req)
    elapsed := time.Since(start)
    if w.Code != http.StatusOK {
        t.Fatalf("dispatch failed: status %d", w.Code)
    }
    const ceiling = 50 * time.Millisecond
    if elapsed > ceiling {
        t.Errorf("2-hop traversal on 10k rows took %s, want <%s (Phase 70 D-07 ceiling)", elapsed, ceiling)
    }
}
```

```yaml
on:
  schedule:
    # 03:00 UTC nightly — after main branch activity settles.
    - cron: '0 3 * * *'

permissions:
  contents: read

jobs:
  traversal-bench:
    name: Phase 70 traversal bench (D-07 gate)
    runs-on: ubuntu-latest
    timeout-minutes: 15
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-go@v5
        with:
          go-version-file: 'go.mod'
          cache: true

      - name: Run bench (n=6)
        env:
          TMPDIR: /tmp/ci
          GOMAXPROCS: '2'  # match shared-cpu-1x / shared-cpu-2x replica class
        run: |
          mkdir -p /tmp/ci
          go test -bench='BenchmarkTraversal' -benchtime=3s -count=6 -run='^$' -benchmem \
            ./internal/pdbcompat/ > /tmp/ci/bench-current.txt

      - name: Install benchstat
        run: go install golang.org/x/perf/cmd/benchstat@latest

      - name: Compare against baseline
        run: |
          if [ -f .planning/phases/70-cross-entity-traversal/bench-baseline.txt ]; then
            benchstat .planning/phases/70-cross-entity-traversal/bench-baseline.txt /tmp/ci/bench-current.txt | tee bench-delta.txt
            # Fail if any line shows >20% regression.
            if grep -qE '\+[2-9][0-9]\.[0-9]+%|\+[0-9]{3,}\.[0-9]+%' bench-delta.txt; then
              echo "::error::Bench regression >20% detected — Phase 70 D-07 gate"
              exit 1
            fi
          else
            echo "No baseline yet — committing current as initial baseline (manual follow-up)"
            cp /tmp/ci/bench-current.txt bench-baseline-suggested.txt
          fi

      - name: Upload bench artefacts
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: traversal-bench-${{ github.run_id }}
          path: |
            /tmp/ci/bench-current.txt
            bench-delta.txt
            bench-baseline-suggested.txt
```
