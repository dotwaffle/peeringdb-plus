```go
func seedFixtures(t *testing.T, c *ent.Client, fixtures []parity.Fixture) {
    t.Helper()
    ctx := context.Background()
    idMap := make(map[string]map[int]int) // entity → fixtureID → persistedID
    // Pass 1: seed without FK refs; record persisted IDs
    for _, f := range fixtures {
        if _, ok := f.Fields["__fk"]; ok { continue }
        persist(ctx, c, f, idMap)
    }
    // Pass 2: seed FK-dependent fixtures, resolving __fk entries
    for _, f := range fixtures {
        if _, ok := f.Fields["__fk"]; !ok { continue }
        persistWithFKs(ctx, c, f, idMap)
    }
}
```
