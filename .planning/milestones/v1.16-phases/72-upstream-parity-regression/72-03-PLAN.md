```go
func TestFixtures_UnicodeSanity(t *testing.T) {
    t.Parallel()
    if len(parity.UnicodeFixtures) < 32 { t.Fatalf("want ≥32; got %d", len(parity.UnicodeFixtures)) }
    var diacritic, cjk bool
    for _, f := range parity.UnicodeFixtures {
        for _, v := range f.Fields {
            s, ok := v.(string)
            if !ok { continue }
            for _, r := range s {
                if r >= 0x4E00 && r <= 0x9FFF { cjk = true }
                if r == 'ü' || r == 'ö' || r == 'é' || r == 'ñ' { diacritic = true }
            }
        }
    }
    if !diacritic { t.Error("no diacritic unicode sample") }
    if !cjk { t.Error("no CJK unicode sample") }
}

func TestFixtures_InSanity(t *testing.T) {
    t.Parallel()
    byID := map[int]bool{}
    var hasSentinel bool
    for _, f := range parity.InFixtures {
        if f.Entity == "network" {
            byID[f.ID] = true
            if f.ID == 999999 { hasSentinel = true }
        }
    }
    if !hasSentinel { t.Error("no empty-__in sentinel fixture") }
    for id := 100000; id <= 105000; id++ {
        if !byID[id] { t.Fatalf("missing network fixture ID %d", id); }
    }
}

func TestFixtures_TraversalSanity(t *testing.T) {
    t.Parallel()
    var twoHop, silentIgnore bool
    for _, f := range parity.TraversalFixtures {
        if h, ok := f.Fields["__hop"].(int); ok && h == 2 { twoHop = true }
        if o, ok := f.Fields["__expected_outcome"].(string); ok && o == "silent-ignore" { silentIgnore = true }
    }
    if !twoHop { t.Error("no 2-hop fixture") }
    if !silentIgnore { t.Error("no silent-ignore fixture (TRAVERSAL-04)") }
}
```
