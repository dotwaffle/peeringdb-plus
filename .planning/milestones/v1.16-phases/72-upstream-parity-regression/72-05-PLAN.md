```go
// Package parity benchmarks.
//
// CI workflow: these benchmarks run on pushes to main via a benchstat-comparing
// workflow (not per-PR — benchmarks are noisy and would block merges). Local
// dev command: `go test -run=^$ -bench=BenchmarkParity -benchtime=5x -count=6
// ./internal/pdbcompat/parity/ | benchstat -`.
//
// All benchmarks use b.Loop() per GO-TOOL-1. Reference precedent:
// internal/pdbcompat/projection_bench_test.go (Phase 46).
package parity_test

import (
    "context"
    "strconv"
    "strings"
    "testing"

    "github.com/dotwaffle/peeringdb-plus/internal/testutil"
    parityfx "github.com/dotwaffle/peeringdb-plus/internal/testutil/parity"
)

// BenchmarkParity_TwoHopTraversal: /api/ixpfx?ixlan__ix__id=N — direct-edge 2-hop.
// Upstream citation: pdb_api_test.py:5081. Locks Phase 70 D-04 cost ceiling.
func BenchmarkParity_TwoHopTraversal(b *testing.B) {
    b.ReportAllocs()
    c := testutil.SetupClient(b)
    srv := newTestServer(b, c)
    seedFixtures(b, c, parityfx.TraversalFixtures)
    path := "/api/ixpfx?ixlan__ix__id=20"
    b.ResetTimer()
    for b.Loop() {
        status, body := httpGet(b, srv, path)
        if status != 200 { b.Fatalf("want 200; got %d", status) }
        _ = body
    }
}

// BenchmarkParity_LimitZeroStreaming: exercises Phase 71 stream.go on a 5000-row
// network population with ?limit=0 (unbounded). Locks the streaming invariant —
// if someone re-introduces full-slice materialisation, allocs/op will jump 10x+.
func BenchmarkParity_LimitZeroStreaming(b *testing.B) {
    b.ReportAllocs()
    c := testutil.SetupClient(b)
    srv := newTestServer(b, c)
    seedNetworkBlock(b, c, 200000, 5000)
    b.ResetTimer()
    for b.Loop() {
        status, body := httpGet(b, srv, "/api/net?limit=0")
        if status != 200 { b.Fatalf("want 200; got %d", status) }
        if len(body) < 100_000 { b.Fatalf("truncated body: %d bytes", len(body)) }
    }
}

// BenchmarkParity_InFiveThousandElements: ?id__in=<5001 csv ids> — exercises the
// Phase 69 json_each(?) single-bind rewrite. Regression would trip SQLite's
// 999-variable limit and fail outright.
func BenchmarkParity_InFiveThousandElements(b *testing.B) {
    b.ReportAllocs()
    c := testutil.SetupClient(b)
    srv := newTestServer(b, c)
    seedFixtures(b, c, parityfx.InFixtures)

    ids := make([]string, 0, 5001)
    for i := 100000; i <= 105000; i++ { ids = append(ids, strconv.Itoa(i)) }
    path := "/api/net?id__in=" + strings.Join(ids, ",")

    b.ResetTimer()
    for b.Loop() {
        status, _ := httpGet(b, srv, path)
        if status != 200 { b.Fatalf("want 200; got %d", status) }
    }
}

// seedNetworkBlock is a benchmark-local helper to synthesise N sequential
// network rows without coupling to fixture files. Used by LimitZeroStreaming.
func seedNetworkBlock(tb testing.TB, c *ent.Client, startID, n int) {
    tb.Helper()
    ctx := context.Background()
    for i := 0; i < n; i++ {
        if err := c.Network.Create().
            SetID(startID+i).
            SetName("bench-net-" + strconv.Itoa(i)).
            SetAsn(40000 + i).
            Exec(ctx); err != nil {
            tb.Fatalf("seed: %v", err)
        }
    }
}
```
