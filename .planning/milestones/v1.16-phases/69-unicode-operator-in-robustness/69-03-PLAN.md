```go
func upsertOrganizations(ctx context.Context, tx *ent.Tx, orgs []peeringdb.Organization) ([]int, error) {
    return upsertBatch(ctx, orgs,
        func(o peeringdb.Organization) int { return o.ID },
        func(o peeringdb.Organization) *ent.OrganizationCreate {
            b := tx.Organization.Create().
                SetID(o.ID).
                SetName(o.Name).
                SetAka(o.Aka).
                SetNameLong(o.NameLong).
                SetWebsite(o.Website).
                ...
                SetCity(o.City).
                ...
            return b
        },
        func(ctx context.Context, builders []*ent.OrganizationCreate) error {
            return tx.Organization.CreateBulk(builders...).
                OnConflict(...).
                UpdateNewValues().
                Exec(ctx)
        },
        "organizations",
    )
}
```

The extension: add `.SetNameFold(unifold.Fold(o.Name))`, `.SetAkaFold(unifold.Fold(o.Aka))`, `.SetCityFold(unifold.Fold(o.City))` in the builder closure. The `OnConflict` / `UpdateNewValues` path re-runs on re-sync — which means re-sync OVERWRITES the `_fold` columns. That's the desired behaviour: if a PeeringDB row's `name` changes from `Zurich` to `Zürich`, the next sync cycle re-computes `name_fold = "zurich"` either way.
