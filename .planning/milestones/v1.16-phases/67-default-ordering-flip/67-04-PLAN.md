    if token == "" { return 0, nil }

```go
type streamCursor struct {
    Updated time.Time
    ID      int
}

func (c streamCursor) empty() bool { return c.Updated.IsZero() && c.ID == 0 }

func encodeStreamCursor(c streamCursor) string {
    if c.empty() { return "" }
    body := fmt.Sprintf("%s:%d", c.Updated.UTC().Format(time.RFC3339Nano), c.ID)
    return base64.StdEncoding.EncodeToString([]byte(body))
}

func decodeStreamCursor(token string) (streamCursor, error) {
    if token == "" { return streamCursor{}, nil }
    raw, err := base64.StdEncoding.DecodeString(token)
    if err != nil { return streamCursor{}, fmt.Errorf("decode stream cursor: %w", err) }
    s := string(raw)
    idx := strings.LastIndex(s, ":")
    if idx < 0 { return streamCursor{}, fmt.Errorf("invalid stream cursor body: %q", s) }
    t, err := time.Parse(time.RFC3339Nano, s[:idx])
    if err != nil { return streamCursor{}, fmt.Errorf("parse stream cursor timestamp: %w", err) }
    id, err := strconv.Atoi(s[idx+1:])
    if err != nil { return streamCursor{}, fmt.Errorf("parse stream cursor id: %w", err) }
    if id < 0 { return streamCursor{}, fmt.Errorf("invalid stream cursor: negative id %d", id) }
    return streamCursor{Updated: t, ID: id}, nil
}
```

## Trust Boundaries

Base64-encoded cursor is untrusted user input when passed in `page_token` field of a List/Stream request. `decodeStreamCursor` validates base64, RFC3339Nano timestamp format, decimal integer id, and rejects negative ids. No SQL injection risk (cursor values flow through ent typed predicates, not string concatenation).
