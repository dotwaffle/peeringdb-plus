1. **Keep** `parseRetryAfter` (already correct ‚Äî handles numeric and HTTP-date per RFC 9110 ¬ß10.2.3).

```go
package sync

import (
    "context"
    "encoding/json"
    "fmt"
    "log/slog"
    "net/url"

    "github.com/dotwaffle/peeringdb-plus/ent"
    "github.com/dotwaffle/peeringdb-plus/internal/otel"  // for metrics
    "github.com/dotwaffle/peeringdb-plus/internal/peeringdb"
    otelattr "go.opentelemetry.io/otel/attribute"
    "go.opentelemetry.io/otel/metric"
)

// fkBackfillKey is the dedup-cache key. typeName + id pair. Per-cycle scope.
type fkBackfillKey struct {
    Type string
    ID   int
}

// fkBackfillResult mirrors the otel attribute "result". Used as both a return
// signal AND a metric label. Values: "hit", "miss", "ratelimited", "error".
type fkBackfillResult string

const (
    fkBackfillHit         fkBackfillResult = "hit"
    fkBackfillMiss        fkBackfillResult = "miss"
    fkBackfillRateLimited fkBackfillResult = "ratelimited"
    fkBackfillError       fkBackfillResult = "error"
)

// fkBackfillParent attempts to fetch a missing parent from upstream and insert
// it (as a tombstoned or live row, preserving upstream status). Returns true
// iff the row is now present in the DB and the child can be linked.
//
// Caller MUST have already established that the parent is absent locally
// (fkHasParent returned false). This function:
//   1. Checks per-cycle dedup cache; on hit, returns the cached result.
//   2. Checks per-cycle cap (PDBPLUS_FK_BACKFILL_MAX_PER_CYCLE); cap-hit ‚Üí ratelimited.
//   3. Fetches /api/<type>?since=1&id__in=<id> (since=1 ensures deleted rows surface).
//   4. On 200 + non-empty data: inserts via single<Type>Upsert; result=hit.
//   5. On 200 + empty data (truly absent upstream): result=miss.
//   6. On error: result=error.
//
// Cache and counter are updated for every code path. fkBackfillTried is
// reset at the top of Worker.Sync alongside cycleStart.
func (w *Worker) fkBackfillParent(ctx context.Context, tx *ent.Tx, parentType string, parentID int) bool {
    key := fkBackfillKey{Type: parentType, ID: parentID}

    if _, tried := w.fkBackfillTried[key]; tried {
        // Already tried this cycle ‚Äî re-check DB rather than re-fetch.
        return w.dbHasRecord(ctx, tx, parentType, parentID)
    }
    w.fkBackfillTried[key] = struct{}{}

    if w.fkBackfillCount >= w.fkBackfillCap {
        w.recordBackfill(ctx, parentType, fkBackfillRateLimited)
        w.logger.LogAttrs(ctx, slog.LevelWarn, "fk backfill cap reached",
            slog.String("type", parentType),
            slog.Int("id", parentID),
            slog.Int("cap", w.fkBackfillCap))
        return false
    }
    w.fkBackfillCount++

    raw, fetchErr := w.fetchSingleByID(ctx, parentType, parentID)
    if fetchErr != nil {
        w.recordBackfill(ctx, parentType, fkBackfillError)
        w.logger.LogAttrs(ctx, slog.LevelWarn, "fk backfill fetch failed",
            slog.String("type", parentType), slog.Int("id", parentID),
            slog.Any("error", fetchErr))
        return false
    }
    if len(raw) == 0 {
        w.recordBackfill(ctx, parentType, fkBackfillMiss)
        w.logger.LogAttrs(ctx, slog.LevelDebug, "fk backfill: parent absent upstream",
            slog.String("type", parentType), slog.Int("id", parentID))
        return false
    }

    if upsertErr := w.upsertSingleRaw(ctx, tx, parentType, raw); upsertErr != nil {
        w.recordBackfill(ctx, parentType, fkBackfillError)
        w.logger.LogAttrs(ctx, slog.LevelWarn, "fk backfill upsert failed",
            slog.String("type", parentType), slog.Int("id", parentID),
            slog.Any("error", upsertErr))
        return false
    }

    // Mirror the registry so subsequent same-cycle children find the parent
    // in memory rather than re-querying the DB.
    w.fkRegisterIDs(parentType, []int{parentID})

    w.recordBackfill(ctx, parentType, fkBackfillHit)
    w.logger.LogAttrs(ctx, slog.LevelInfo, "fk backfill: parent inserted",
        slog.String("type", parentType), slog.Int("id", parentID))
    return true
}

// fetchSingleByID issues a list query restricted to one ID with since=1 to
// catch tombstones. Returns the single raw JSON object or nil if upstream
// returned an empty array.
func (w *Worker) fetchSingleByID(ctx context.Context, typeName string, id int) (json.RawMessage, error) {
    // Use the existing FetchAll machinery ‚Äî it owns rate limiting, retries,
    // and the standardized error path. ?since=1&id__in=<id> guarantees
    // status=deleted rows surface; the per-row id__in filter caps page count
    // at 1 page.
    q := url.Values{}
    q.Set("since", "1")
    q.Set("id__in", fmt.Sprintf("%d", id))
```

```go
// and returns each top-level data element as a json.RawMessage clone. Used
// by sync's FK backfill which fetches one row at a time.
func (c *Client) FetchRaw(ctx context.Context, objectType string, params url.Values) ([]json.RawMessage, error) {
    u := fmt.Sprintf("%s/api/%s", c.baseURL, objectType)
    if len(params) > 0 {
        u += "?" + params.Encode()
    }
    resp, err := c.doWithRetry(ctx, u)
    if err != nil { return nil, err }
    defer resp.Body.Close()
    var env struct{ Data []json.RawMessage `json:"data"` }
    if err := json.NewDecoder(resp.Body).Decode(&env); err != nil {
        return nil, fmt.Errorf("decode %s: %w", u, err)
    }
    return env.Data, nil
}
```
