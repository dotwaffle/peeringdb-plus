```go
// sdkmetric.NewView signature
func NewView(criteria Instrument, mask Stream) View

// Drop aggregation — produces no exported data
type AggregationDrop struct{}

// Histogram coarsening
type AggregationExplicitBucketHistogram struct {
    Boundaries []float64
    NoMinMax   bool
}

// Stream allows attribute filtering via Stream.AttributeFilter func(KeyValue) bool
type Stream struct {
    Name            string
    Description     string
    Unit            string
    Aggregation     Aggregation
    AttributeFilter attribute.Filter
}

// attribute.NewAllowKeysFilter returns an attribute.Filter that keeps ONLY
// the listed keys, dropping all others. Use for the http.method drop +
// http.response.status_code retain pattern.
func NewAllowKeysFilter(keys ...attribute.Key) Filter
```

```go
),
```
